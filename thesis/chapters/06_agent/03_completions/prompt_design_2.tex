\newcommand{\itemizethis}[1]{%
	\if\relax\detokenize{#1}\relax%
		% do nothing
	\else%
		\setlength{\leftmargini}{.3cm}%
		\begin{varwidth}[t]{\linewidth}%
		\begin{itemize}[topsep=0pt,parsep=0pt]
			\item #1
		\end{itemize}%
		\end{varwidth}%
	\fi%
}
